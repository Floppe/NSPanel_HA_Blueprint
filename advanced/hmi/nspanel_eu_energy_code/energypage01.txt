                // slider2 at the house, waiting for slider1 to finish
                solar_slider1.val--
                if(solar_slider1.val==solar_slider1.minval)
                {
                    // jump back to the sun
                    solar_slider1.val=solar_slider1.maxval
                    // start slider2
                    solar_slider2.val=solar_slider2.minval
                }
            }
